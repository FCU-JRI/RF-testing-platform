本次實地測試在視距（LOS）條件下評估航電鏈路的可靠性。 透過將定向八木天線架設於固定地面站，並在冗餘航電端配戴全向鞭狀天線，進行不同偏角下的性能測量。

地面接收/發射端 (社長家)

配置 433 MHz & 915 MHz 八木天線 (Yagi) + Orion v5 核心板，執行定向高增益收發。

航電/冗餘模擬端 (望高寮)

配置鞭型天線 + Orion v6 冗餘系統板 / 航電主系統板，模擬在空中的全向輻射情況。

1

韌體燒錄與頻率配置

確保兩端天線皆已牢固接妥，並使用 rf_test_manager.py 管理程式自動燒錄。

  • 開啟系統終端機並執行自動化管理器腳本：
python rf_test_manager.py
  • 選擇 2) Flash Firmware 進入燒錄選單。
  • 根據組別選擇發射端 (Sender) / 接收端 (Receiver)、硬體版本 (Orion v5/v6)，以及載波頻率 (433MHz/915MHz)。
  • 程式會自動更動代碼中的 LORA_FREQ 巨集並調用 PlatformIO CLI 進行燒錄。
2

對位預測量與天線指向優化

為了獲得最大信號增益，必須在測試前進行環境背景訊號測量與天線對位校準。

  • 接收端：在選單中啟動 4) Run Receiver Data Logger 進入即時封包監聽模式。
  • 發射端：在選單中啟動 3) Start Transmitter，並輸入 3 (Pre-Test 模式，發射 SF7、間隔 1 秒)。
  • 接收端會顯示動態的 RSSI 強度，根據該數值微調八木天線指向以達最佳訊號（最接近 0 dBm）。
  • 對齊完成後，發射端輸入 x 停止預測。
3

常規 6 次測試紀錄 (No. 1 - No. 6)

進行常規無偏角 0° 的 6 輪對稱性傳輸，每一輪包含發射 100 個封包。

  • 接收端保持 Data Logger 監聽，並可隨時在終端機輸入 r 重置統計計數。
  • 發射端選擇 1) Formal Test（正式測試），輸入 7 (Spreading Factor = 7) 開始傳輸。
  • 傳輸完成後，rf_test_manager.py 會自動將記錄的 RSSI, SNR, 丟包率以 CSV 格式寫入至 logs/ 資料夾。
  • 重置接收端 (輸入 r)，並重複執行共 6 次以完成常規矩陣。
4

極化偏角測試

測試八木天線在極化平面旋轉 30° 與 45° 角度下的抗干擾及訊號衰減表現。

  • 調整望高寮端的鞭狀天線/社長家端的八木天線夾角至 30° 或 45°。
  • 接收端重置統計，發射端以相同程序發射 100 個 SF7 封包。
  • 測試完畢後，將 Logger 收集到的平均數值填入上方「交互測試矩陣」對應欄位中存檔。
#include <SPI.h>
#include <LoRa.h>

// =================================================================
// 測試載波頻率配置（請依據當下測試配置切換註解，再進行燒錄）
// =================================================================
//#define LORA_FREQ   433E6   // 433 MHz 鏈路：配置 3 & 配置 4
#define LORA_FREQ      915E6   // 915 MHz 鏈路：配置 1 & 配置 2

// =================================================================
// ESP32 硬體腳位配置（發射端 Orion v5 自訂 SPI 腳位）
// =================================================================
#define SCK_PIN   18
#define MISO_PIN  19
#define MOSI_PIN  23
#define NSS_PIN   33   // CS 片選 (Orion v5)
#define RST_PIN   26   // Reset (Orion v5)
#define DIO0_PIN  13   // 中斷腳位 (Orion v5)

enum Mode { IDLE, PRE_TEST, FORMAL_TEST, STRESS_TEST };
Mode currentMode = IDLE;
int currentSF = 7;
unsigned long testInterval = 500; 
unsigned long packetCounter = 0;
unsigned long lastSendTime = 0;
const int packetLimit = 100;
int targetPayloadLength = 256; // 配合紀錄表，預設改為 256 Bytes
String currentUUID = "N/A";    // 儲存當前測試的 UUID

void handleSerial();
void sendPacket();
void updateParams(int sf);
void printMenu();
unsigned long getSafeInterval(int sf); 

void setup() {
  Serial.begin(115200);
  while (!Serial && millis() < 3000);
  delay(500);
  while (Serial.available() > 0) {
    Serial.read();
  }
  Serial.setTimeout(50);
  
  Serial.print("\n=== LoRa 發射端初始化 [頻率: ");
  Serial.print(LORA_FREQ / 1E6);
  Serial.println(" MHz] ===");

  SPI.begin(SCK_PIN, MISO_PIN, MOSI_PIN, NSS_PIN);
  LoRa.setPins(NSS_PIN , RST_PIN, DIO0_PIN);
  
  if (!LoRa.begin(LORA_FREQ)) { 
    Serial.println("LoRa 初始化失敗，請確認接線與頻率設定！"); 
    while(1); 
  }
  
  LoRa.setSignalBandwidth(125E3);
  LoRa.setCodingRate4(6);
  LoRa.setPreambleLength(12);
  LoRa.setSyncWord(0xF1);
  LoRa.enableCrc(); // 【盲點修復】開啟硬體 CRC
  LoRa.setTxPower(20); // 【盲點修復】強制設定最大功率
  printMenu();
}

void loop() {
  handleSerial();

  if (currentMode != IDLE) {
    if (millis() - lastSendTime > testInterval) {
      sendPacket();
      lastSendTime = millis(); 
      if ((currentMode == FORMAL_TEST || currentMode == STRESS_TEST) && packetCounter >= packetLimit) {
        Serial.println("\n>>> 測試任務完成，回到待機。");
        currentMode = IDLE;
        printMenu();
      }
    }
  }
}

void handleSerial() {
  if (Serial.available() > 0) {
    String input = Serial.readStringUntil('\n');
    input.trim();
    String lowerInput = input;
    lowerInput.toLowerCase(); 
    
    if (lowerInput == "p") {
      currentMode = PRE_TEST;
      testInterval = 1000;
      updateParams(7);
      Serial.println("\n>>> [環境預測量階段] SF7, Interval 1s");
      lastSendTime = millis() - testInterval; 
    } 
    else if (lowerInput == "x") {
      currentMode = IDLE;
      Serial.println("\n>>> 已停止。");
    } 
    else if (lowerInput.startsWith("u ")) {
      currentUUID = input.substring(2);
      currentUUID.trim();
      Serial.print("\n>>> [設定] UUID 已更改為: ");
      Serial.println(currentUUID);
    } 
    else if (lowerInput.startsWith("s ")) {
      int sf = 0, inter = 0;
      if (sscanf(lowerInput.c_str(), "s %d %d", &sf, &inter) == 2) {
        if (sf >= 6 && sf <= 12 && inter > 0) {
          currentMode = STRESS_TEST;
          testInterval = inter;
          updateParams(sf);
          Serial.print("\n>>> [極限壓力測試] SF"); Serial.print(sf);
          Serial.print(", Interval "); Serial.print(inter); Serial.println("ms");
          lastSendTime = millis() - testInterval; 
        }
      }
    } 
    else if (lowerInput.startsWith("l ")) {
      int len = 0;
      if (sscanf(lowerInput.c_str(), "l %d", &len) == 1) {
        if (len >= 10 && len <= 256) {
          targetPayloadLength = len;
          Serial.print("\n>>> [設定] 目標封包長度已更改為: "); 
          Serial.print(targetPayloadLength); Serial.println(" Bytes");
        }
      }
    } 
    else {
      int sf = input.toInt();
      if (sf >= 6 && sf <= 12) {
        currentMode = FORMAL_TEST;
        updateParams(sf);
        testInterval = getSafeInterval(sf);
        Serial.print("\n>>> [正式測試階段] SF"); Serial.print(sf);
        Serial.print(", Interval "); Serial.print(testInterval); Serial.println("ms");
        lastSendTime = millis() - testInterval; 
      }
    }
  }
}

void updateParams(int sf) {
  currentSF = sf;
  packetCounter = 0;
  LoRa.setSpreadingFactor(sf);
}

void sendPacket() {
  unsigned long start = millis();
  String payload = "";
  if (currentMode == PRE_TEST) payload += "TST:";
  else if (currentMode == STRESS_TEST) payload += "STR:";
  else payload += "FRM:";
  
  payload += String(packetCounter);
  payload += ":";
  payload += currentUUID;

  while (payload.length() < (size_t)targetPayloadLength) {
    payload += "*"; 
  }

  LoRa.beginPacket();
  LoRa.print(payload);
  LoRa.endPacket(); 
  unsigned long duration = millis() - start;

  Serial.print(currentMode == PRE_TEST ? "[PRE]" : (currentMode == STRESS_TEST ? "[STRESS]" : "[FORM]"));
  Serial.print(" SF"); Serial.print(currentSF);
  Serial.print(" | ID:"); Serial.print(packetCounter);
  Serial.print(" | UUID:"); Serial.print(currentUUID);
  Serial.print(" | Len:"); Serial.print(payload.length()); 
  Serial.print(" | ToA:"); Serial.print(duration); Serial.println("ms");
  packetCounter++;
}

unsigned long getSafeInterval(int sf) {
  if (sf <= 7) return 250;
  if (sf == 8) return 500;
  if (sf == 9) return 1000;
  if (sf == 10) return 2000;
  return 5000;
}

void printMenu() {
  Serial.println("\n--- 控制指令 ---");
  Serial.println("  l [Len]    : 設定封包長度 (預設 256 Bytes)");
  Serial.println("  u [UUID]   : 設定當前測試 UUID");
  Serial.println("  p          : 環境測試(SF7 慢速發送)");
  Serial.println("  6-12       : 正式測試開始(限制發射 100包)");
  Serial.println("  s [SF] [I] : 壓力測試(自定義 SF 與發射頻率)");
  Serial.println("  x          : 停止發射");
}

  while (!Serial && millis() < 3000);
  delay(500);
  while (Serial.available() > 0) {
    Serial.read();
  }
